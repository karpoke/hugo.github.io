$$
\large\begin{align*}
\frac{1}{3} &= 0.\stackrel{\frown}{3} \\
\frac{1}{3} &= 0.333\dots \\
3 \cdot \frac{1}{3} &= 3 \cdot 0.333\dots \\
1 &= 0.999\dots \\
1 &= 0.\stackrel{\frown}{9}
\end{align*}
$$
